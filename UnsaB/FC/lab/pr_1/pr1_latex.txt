\chapter{1.2 Velocidad constante en dos dimensiones}

\begin{itemize}
    \item 1)
    \begin{lstlisting}
import matplotlib.pyplot as plt
import numpy as np
h=0.01
x=5
y=4
vx=-2
vy=0
ax=0
ay=0
px=[]
py=[]
pv=[]
pa=[]
pt = np.arange(2,5,h)

for t in pt:
	x=x+vx*h
	y=y+vy*h
	vx=vx+ax*h
	vy=vy+ay*h
	px.append(x)
	py.append(y)
	pv.append([vx,vy])
	pa.append([ax,ay])

    \end{lstlisting}

import numpy as np
h=0.01
